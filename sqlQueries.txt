#Selecting Two or More Columns From One Table and Filtering Using Numeric Comparison in WHERE 

Comparison Operator	        Description

=	                     Is equal to
>	                     Is greater than

<	                     Is less than
>=	                     Is greater than or equal to
<=	                     Is less than or equal to
<>	                     Is not equal to


#Using the = (Is equal to) Operator
select  column_name1 from table_name WHERE column_name1 = 'name';

#Using the > (Is greater than) Operator
select column_name1, column_name2 from table_name WHERE salary > 3000;

#Using the < (Is less than) Operator
select column_name1, column_name2 from table_name WHERE salary < 3000;

#Using the >= (Is greater than or equal to) Operator
select column_name1, column_name2 from table_name WHERE salary >= 3000;

#Using the <= (Is less than or equal to) Operator
select column_name1, column_name2 from table_name WHERE salary <= 3000;

#Using the <> (Is not equal to) Operator
select column_name1, column_name2 from table_name WHERE salary < 3000;
